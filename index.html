  aes(x = rounds, y = cumsum(correct)) +
  geom_line(linewidth = 1) +
  scale_x_continuous(breaks = seq(min(rounds), max(rounds), 1)) +
  scale_y_continuous(breaks = seq(0, total_correct, 10)) +
